prog EX01-comando enquanto
real s,n,d
s<-0;
n<-1;
d<-1;
enquanto(d<=50)
{
    s<-s+n/d;
    n<-n+2;
    d<-d+1;
}
imprima"somatoria",s;
fim prog

prog EX01-comado para
real s,n,d;
s<-0;
n<-1;
para(d<-;d<=50;d<-d+1)
{
    s=s+n/d;
    n=n+2;
}
imprima"somatoria:",s;
